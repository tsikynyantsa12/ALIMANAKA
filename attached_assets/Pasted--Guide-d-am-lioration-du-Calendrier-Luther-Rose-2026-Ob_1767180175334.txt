def hex_to_rgb(hex_color):
    """Convertir couleur hexadÃ©cimale en RGB"""
    hex_color = hex_color.lstrip('#')
    return tuple(int(hex_color[i:i+2], 16) for i in (0, 2, 4))

def hex_to_rgba(hex_color, alpha=255):
    """Convertir hex en RGBA avec transparence"""
    rgb = hex_to_rgb(hex_color)
    return rgb + (alpha,)

def draw_wave_decoration(draw, width, height, color, position='top'):
    """Dessiner une vague dÃ©corative moderne"""
    import math
    
    wave_height = 120
    wave_amplitude = 40
    
    if position == 'top':
        y_base = wave_height
        direction = 1
    else:
        y_base = height - wave_height
        direction = -1
    
    # CrÃ©er les points de la vague
    points = []
    for x in range(width + 10):
        # Formule de vague sinusoÃ¯dale
        y = y_base + direction * wave_amplitude * math.sin(x * 0.01)
        points.append((x, y))
    
    # ComplÃ©ter le polygone
    if position == 'top':
        points.append((width, 0))
        points.append((0, 0))
    else:
        points.append((width, height))
        points.append((0, height))
    
    # Dessiner la vague
    draw.polygon(points, fill=hex_to_rgb(color))
```

def draw_checkered_background(img, cell_size=20):
    """CrÃ©er un fond en damier transparent"""
    draw = ImageDraw.Draw(img)
    width, height = img.size
    
    # Couleurs du damier
    color1 = (220, 220, 220)  # Gris clair
    color2 = (255, 255, 255)  # Blanc
    
    for y in range(0, height, cell_size):
        for x in range(0, width, cell_size):
            # Alterner les couleurs
            color = color1 if (x // cell_size + y // cell_size) % 2 == 0 else color2
            draw.rectangle([x, y, x + cell_size, y + cell_size], fill=color)
```

def draw_calendar_grid(draw, year, month, x_start, y_start, 
                       cell_width, cell_height, primary_color):
    """Dessiner la grille du calendrier"""
    DAY_NAMES = ["L", "M", "M", "J", "V", "S", "D"]
    
    # En-tÃªtes des jours
    for i, day in enumerate(DAY_NAMES):
        x = x_start + i * cell_width + cell_width // 2
        bbox = font_medium.getbbox(day)
        text_width = bbox[2] - bbox[0]
        draw.text((x - text_width // 2, y_start), 
                 day, fill=hex_to_rgb(primary_color), font=font_medium)
    
    # Ligne sous les en-tÃªtes
    y_line = y_start + 35
    draw.line([x_start, y_line, x_start + 7 * cell_width, y_line], 
              fill=hex_to_rgb(primary_color), width=2)
    
    # Grille des dates
    cal = calendar.monthcalendar(year, month)
    y = y_line + 10
    
    for week in cal:
        for i, day in enumerate(week):
            if day == 0:
                continue
            
            x = x_start + i * cell_width
            
            # Fond pour dimanche
            if i == 6:
                draw.rectangle([x+2, y+2, x+cell_width-2, y+cell_height-2], 
                             fill=hex_to_rgb(COLORS['light_gray']))
            
            # NumÃ©ro du jour
            day_str = str(day)
            bbox = font_large.getbbox(day_str)
            text_width = bbox[2] - bbox[0]
            draw.text((x + (cell_width - text_width) // 2, y + 10), 
                     day_str, fill=hex_to_rgb(COLORS['black']), 
                     font=font_large)
            
            # Bordure de cellule
            draw.rectangle([x, y, x + cell_width, y + cell_height], 
                         outline=hex_to_rgb(primary_color), width=1)
        
        y += cell_height
